async function readLoop() { while (isConnected && device && device.opened) { try { const result = await device.transferIn(epIn, 64); if (result.data && result.data.byteLength > 0) { const text = decoder.decode(result.data).trim(); if (text) { // CHANGE: Convert string to a float and format to 1 decimal place const floatVal = parseFloat(text); if (!isNaN(floatVal)) { displayEl.innerText = floatVal.toFixed(1); } // Echo back the original text await device.transferOut(epOut, encoder.encode(text + "\n")); } } } catch (e) { if (isConnected) { console.error("Read Error:", e); disconnect(); } break; } } }
